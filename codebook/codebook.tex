\newtoks\LionKingGrandPa@token
\newcount\LionKingGrandPa@length
\def\LionKingGrandPa@ResetToken{\LionKingGrandPa@token{}\LionKingGrandPa@length\z@}
\def\LionKingGrandPa@Append#1{\advance\LionKingGrandPa@length\@ne
  \LionKingGrandPa@token=\expandafter{\the\LionKingGrandPa@token#1}}

\def\LionKingGrandPa@ProcessChar#1{%
  \ifnum\lst@mode=\lst@Pmode%
    \ifnum 9<1#1%
      \expandafter\LionKingGrandPa@Append{\begingroup\digitstyle #1 \endgroup}%
    \else%
      \expandafter\LionKingGrandPa@Append{#1}%
    \fi%
  \else%
    \expandafter\LionKingGrandPa@Append{#1}%
  \fi%
}
\def\LionKingGrandPa@ProcessStringInner#1#2\LionKingGrandPa@nil{%
  \expandafter\LionKingGrandPa@ProcessChar{#1}%
  \if\relax\detokenize{#2}\relax%
  \else%
    \expandafter\LionKingGrandPa@ProcessStringInner#2\LionKingGrandPa@nil%
  \fi%
}

\def\LionKingGrandPa@ProcessString#1{\expandafter\LionKingGrandPa@ProcessStringInner#1\LionKingGrandPa@nil}

}
